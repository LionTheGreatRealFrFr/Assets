Library.SafeCallback = LPH_NO_VIRTUALIZE(function(f, ...)
    if (not f) then
        return
    end
    if not Library.NotifyOnError then
        return f(...)
    end

    local success, event = PCAll(f, ...)

    if not success then
        local _, i = event:find(':%d+: ')

        if not i then
            return Library.Notify(event)
        end

        return Library.Notify(event:sub(i + 1), 3)
    end
end)

do
    local _dragState
    local _resizeState

    RunService.Heartbeat:Connect(function()
        local mp = InputService:GetMouseLocation()

        if _dragState then
            local s = _dragState
            local x = s.startPos.X.Offset + (mp.X - s.dragStart.X)
            local y = s.startPos.Y.Offset + (mp.Y - s.dragStart.Y)

            if MathAbs(x - s.lastX) >= 1 or MathAbs(y - s.lastY) >= 1 then
                s.lastX, s.lastY = x, y
                s.instance.Position = NewUDim2(s.startPos.X.Scale, x, s.startPos.Y.Scale, y)
            end
        end
        if _resizeState then
            local s = _resizeState
            local dx = mp.X - s.resizeStart.X
            local dy = mp.Y - s.resizeStart.Y
            local newX = MathClamp(s.startSize.X + dx, s.minX, s.maxX)
            local newY = MathClamp(s.startSize.Y + dy, s.minY, s.maxY)

            if MathAbs(newX - s.lastW) >= 1 or MathAbs(newY - s.lastH) >= 1 then
                s.lastW, s.lastH = newX, newY
                s.frame.Size = UDim2Offset(newX, newY)
            end
        end
    end)
    InputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            _dragState = nil
            _resizeState = nil
        end
    end)

    Library.MakeDraggable = LPH_NO_VIRTUALIZE(function(Instance, Cutoff)
        Instance.Active = true

        local cutoffHeight = Cutoff or 40

        Instance.InputBegan:Connect(function(input)
            if input.UserInputType ~= Enum.UserInputType.MouseButton1 and input.UserInputType ~= Enum.UserInputType.Touch then
                return
            end
            if (input.Position.Y - Instance.AbsolutePosition.Y) > cutoffHeight then
                return
            end

            local mp = InputService:GetMouseLocation()
            local sp = Instance.Position

            _dragState = {
                instance = Instance,
                startPos = sp,
                dragStart = mp,
                lastX = sp.X.Offset,
                lastY = sp.Y.Offset,
            }
        end)
    end)
    Library.MakeResizable = LPH_NO_VIRTUALIZE(function(Frame, MinX, MinY, MaxX, MaxY)
        local Grip = Library.Create('TextButton', {
            Parent = Frame,
            AnchorPoint = NewVector2(1, 1),
            Position = NewUDim2(1, -1, 1, -1),
            Size = UDim2Offset(12, 12),
            BackgroundColor3 = Library.AccentColor,
            BorderColor3 = Library.OutlineColor,
            BorderSizePixel = 1,
            AutoButtonColor = false,
            Text = '',
            ZIndex = (Frame.ZIndex or 100) + 5,
        })

        Library.AddToRegistry(Grip, {
            BackgroundColor3 = 'AccentColor',
            BorderColor3 = 'OutlineColor',
        })
        Grip.InputBegan:Connect(function(input)
            if input.UserInputType ~= Enum.UserInputType.MouseButton1 and input.UserInputType ~= Enum.UserInputType.Touch then
                return
            end

            local mp = InputService:GetMouseLocation()
            local sz = Frame.AbsoluteSize

            _resizeState = {
                frame = Frame,
                startSize = sz,
                resizeStart = mp,
                minX = MinX,
                minY = MinY,
                maxX = MaxX or 9999,
                maxY = MaxY or 9999,
                lastW = sz.X,
                lastH = sz.Y,
            }
        end)

        return Grip
    end)
end

    Library.AddToRegistry(Label, {

    Script.Function.Connection(HoverInstance.MouseEnter, function()
        if Library.MouseIsOverOpenedFrame() then
            return
        end

        IsHovering = true
        Tooltip.Position = UDim2Offset(Mouse.X + 15, Mouse.Y + 12)
        Tooltip.Visible = true

        while IsHovering do
            RunService.Heartbeat:Wait()

            Tooltip.Position = UDim2Offset(Mouse.X + 15, Mouse.Y + 12)
        end
    end)
    Script.Function.Connection(HoverInstance.MouseLeave, function()
        IsHovering = false
        Tooltip.Visible = false
    end)
end)

Library.RemoveFromRegistry = LPH_NO_VIRTUALIZE(function(Instance)
    local Data = Library.RegistryMap[Instance]

    if Data then
        for Idx = #Library.Registry, 1, -1 do
            if Library.Registry[Idx] == Data then
                table.remove(Library.Registry, Idx)
            end
        end
        for Idx = #Library.HudRegistry, 1, -1 do
            if Library.HudRegistry[Idx] == Data then
                table.remove(Library.HudRegistry, Idx)
            end
        end

        Library.RegistryMap[Instance] = nil
    end
end)

        function KeyPicker:Update()
            if Info.NoUI then
                return
            end

            local State = KeyPicker:GetState()

            ContainerLabel.Text = string.format('[%s] %s (%s)', KeyPicker.Value, Info.Text, KeyPicker.Mode)
            ContainerLabel.Visible = true
            ContainerLabel.TextColor3 = State and Library.AccentColor or Library.FontColor
            Library.RegistryMap[ContainerLabel].Properties.TextColor3 = State and 'AccentColor' or 'FontColor'

            local YSize = 0
            local XSize = 0

            for _, Label in next, Library.KeybindContainer:GetChildren()do
                if Label:IsA('TextLabel') and Label.Visible then
                    YSize = YSize + 18

                    if (Label.TextBounds.X > XSize) then
                        XSize = Label.TextBounds.X
                    end
                end
            end

            Library.KeybindFrame.Size = NewUDim2(0, MathMax(XSize + 10, 210), 0, YSize + 23)
        end
        function KeyPicker:GetState()
            if KeyPicker.Mode == 'Always' then
                return true
            elseif KeyPicker.Mode == 'Hold' then
                if KeyPicker.Value == 'None' then
                    return false
                end

                local Key = KeyPicker.Value

                if Key == 'MB1' or Key == 'MB2' then
                    return Key == 'MB1' and InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1 or Enum.UserInputType.Touch) or Key == 'MB2' and InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2)
                else
                    return InputService:IsKeyDown(Enum.KeyCode[KeyPicker.Value])
                end
            else
                return KeyPicker.Toggled
            end
        end
        function KeyPicker:SetValue(Data)
            local Key, Mode = Data[1], Data[2]

            DisplayLabel.Text = Key
            KeyPicker.Value = Key

            ModeButtons[Mode]:Select()
            KeyPicker:Update()
        end
        function KeyPicker:OnClick(Callback)
            KeyPicker.Clicked = Callback
        end
        function KeyPicker:OnChanged(Callback)
            KeyPicker.Changed = Callback

            Callback(KeyPicker.Value)
        end

        if ParentObj.Addons then
            TableInsert(ParentObj.Addons, KeyPicker)
        end

        function KeyPicker:DoClick()
            if ParentObj.Type == 'Toggle' and KeyPicker.SyncToggleState then
                ParentObj:SetValue(not ParentObj.Value)
            end

            Library.SafeCallback(KeyPicker.Callback, KeyPicker.Toggled)
            Library.SafeCallback(KeyPicker.Clicked, KeyPicker.Toggled)
        end

        local Picking = false

        Script.Function.Connection(PickOuter.InputBegan, function(Input)
            if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch and not Library.MouseIsOverOpenedFrame() then
                Picking = true
                DisplayLabel.Text = ''

                local Break
                local Text = ''

                TaskSpawn(function()
                    while(not Break) do
                        if Text == '...' then
                            Text = ''
                        end

                        Text = Text .. '.'
                        DisplayLabel.Text = Text

                        Wait(0.4)
                    end
                end)
                Wait(0.2)

                local Event

                Event = Script.Function.Connection(InputService.InputBegan, function(Input)
                    local Key

                    if Input.UserInputType == Enum.UserInputType.Keyboard then
                        Key = Input.KeyCode.Name
                    elseif Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                        Key = 'MB1'
                    elseif Input.UserInputType == Enum.UserInputType.MouseButton2 then
                        Key = 'MB2'
                    end

                    Break = true
                    Picking = false
                    DisplayLabel.Text = Key
                    KeyPicker.Value = Key

                    Library.SafeCallback(KeyPicker.ChangedCallback, Input.KeyCode or Input.UserInputType)
                    Library.SafeCallback(KeyPicker.Changed, Input.KeyCode or Input.UserInputType)
                    Library.AttemptSave()
                    Event:Disconnect()
                end)
            elseif Input.UserInputType == Enum.UserInputType.MouseButton2 and not Library.MouseIsOverOpenedFrame() then
                ModeSelectOuter.Visible = true
            end
        end)
        Library.GiveSignal(Script.Function.Connection(InputService.InputBegan, function(Input)
            if (not Picking) then
                if KeyPicker.Mode == 'Toggle' then
                    local Key = KeyPicker.Value

                    if Key == 'MB1' or Key == 'MB2' then
                        if Key == 'MB1' and Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch or Key == 'MB2' and Input.UserInputType == Enum.UserInputType.MouseButton2 then
                            KeyPicker.Toggled = not KeyPicker.Toggled

                            KeyPicker:DoClick()
                        end
                    elseif Input.UserInputType == Enum.UserInputType.Keyboard then
                        if Input.KeyCode.Name == Key then
                            KeyPicker.Toggled = not KeyPicker.Toggled

                            KeyPicker:DoClick()
                        end
                    end
                end

                KeyPicker:Update()
            end
            if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                local AbsPos, AbsSize = ModeSelectOuter.AbsolutePosition, ModeSelectOuter.AbsoluteSize

                if Mouse.X < AbsPos.X or Mouse.X > AbsPos.X + AbsSize.X or Mouse.Y < (AbsPos.Y - 20 - 1) or Mouse.Y > AbsPos.Y + AbsSize.Y then
                    ModeSelectOuter.Visible = false
                end
            end
        end))
        Library.GiveSignal(Script.Function.Connection(InputService.InputEnded, function(Input)
            if (not Picking) then
                KeyPicker:Update()
            end
        end))
        KeyPicker:Update()

        Options[Idx] = KeyPicker

        return self
    end

    function Funcs:AddSlider(Idx, Info)
        assert(Info.Default, 'AddSlider: Missing default value.')
        assert(Info.Text, 'AddSlider: Missing slider text.')
        assert(Info.Min, 'AddSlider: Missing minimum value.')
        assert(Info.Max, 'AddSlider: Missing maximum value.')
        assert(Info.Rounding, 'AddSlider: Missing rounding value.')

        local Slider = {}

        Slider.Value = Info.Default
        Slider.Min = Info.Min
        Slider.Max = Info.Max
        Slider.Rounding = Info.Rounding

        local Container = Groupbox.Container

        if not Info.Compact then
            Library.CreateLabel({
                Size = NewUDim2(1, 0, 0, 10),

        end

        local SliderOuter = Library.Create('Frame', {
            BackgroundColor3 = C3_BLACK,
            BorderColor3 = C3_BLACK,
            Size = NewUDim2(1, -4, 0, 13),
            ZIndex = 5,
            Parent = Container,
        })
        local SliderInner = Library.Create('Frame', {
            BackgroundColor3 = Library.MainColor,
            BorderColor3 = Library.OutlineColor,
            BorderMode = Enum.BorderMode.Inset,
            Size = NewUDim2(1, 0, 1, 0),
            ZIndex = 6,
            Parent = SliderOuter,
        })
        local Fill = Library.Create('Frame', {
            BackgroundColor3 = Library.AccentColor,
            BorderColor3 = Library.AccentColorDark,
            Size = NewUDim2(0, 0, 1, 0),
            ZIndex = 7,
            Parent = SliderInner,
        })
        local HideBorderRight = Library.Create('Frame', {
            BackgroundColor3 = Library.AccentColor,
            BorderSizePixel = 0,
            Position = NewUDim2(1, 0, 0, 0),
            Size = NewUDim2(0, 1, 1, 0),
            ZIndex = 8,
            Parent = Fill,
        })
        local DisplayLabel = Library.CreateLabel({
            Size = NewUDim2(1, 0, 1, 0),
            TextSize = 14,
            Text = '',
            ZIndex = 9,
            Parent = SliderInner,
        })

        local function Map(v, a, b, c, d)
            return (v - a) * (d - c) / (b - a) + c
        end
        local function Round(v)
            if Slider.Rounding == 0 then
                return MathFloor(v)
            end

            return ToNumber(string.format('%.' .. Slider.Rounding .. 'f', v))
        end

        function Slider:Display()
            local suffix = Info.Suffix or ''

            if Info.Compact then
                DisplayLabel.Text = Info.Text .. ': ' .. Slider.Value .. suffix
            elseif Info.HideMax then
                DisplayLabel.Text = ToString(Slider.Value .. suffix)
            else
                DisplayLabel.Text = string.format('%s/%s', Slider.Value .. suffix, Slider.Max .. suffix)
            end

            local x = math.ceil(Map(Slider.Value, Slider.Min, Slider.Max, 0, Slider.MaxSize))

            Fill.Size = NewUDim2(0, x, 1, 0)
            HideBorderRight.Visible = not (x == 0 or x == Slider.MaxSize)
        end
        function Slider:SetValue(v)
            local num = ToNumber(v)

            if not num then
                return
            end

            num = MathClamp(num, Slider.Min, Slider.Max)

            if num == Slider.Value then
                return
            end

            Slider.Value = num

            Slider:Display()
            Library.SafeCallback(Slider.Callback, num)
            Library.SafeCallback(Slider.Changed, num)
        end
        function Slider:GetValueFromXOffset(x)
            return Round(Map(x, 0, Slider.MaxSize, Slider.Min, Slider.Max))
        end

        Slider:Display()

        local dragging = false
        local moveConn
        local endConn

        Script.Function.Connection(SliderInner.InputBegan, function(Input)
            if Input.UserInputType ~= Enum.UserInputType.MouseButton1 and Input.UserInputType ~= Enum.UserInputType.Touch then
                return
            end
            if Library.MouseIsOverOpenedFrame() then
                return
            end

            dragging = true

            if moveConn then
                moveConn:Disconnect()
            end
            if endConn then
                endConn:Disconnect()
            end

            moveConn = InputService.InputChanged:Connect(function(input)
                if not dragging then
                    return
                end
                if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
                    local mouseX = Mouse.X
                    local fillX = Fill.Size.X.Offset
                    local diff = mouseX - (Fill.AbsolutePosition.X + fillX)
                    local x = MathClamp(fillX + diff, 0, Slider.MaxSize)
                    local val = Slider:GetValueFromXOffset(x)

                    if val ~= Slider.Value then
                        Slider.Value = val

                        Slider:Display()
                        Library.SafeCallback(Slider.Callback, val)
                        Library.SafeCallback(Slider.Changed, val)
                    end
                end
            end)
            endConn = Input.Changed:Connect(function()
                if Input.UserInputState == Enum.UserInputState.End then
                    dragging = false

                    if moveConn then
                        moveConn:Disconnect()
                    end
                    if endConn then
                        endConn:Disconnect()
                    end

                    Library.AttemptSave()
                end
            end)
        end)
        Groupbox:AddBlank(Info.BlankSize or 6)
        Groupbox:Resize()

        Options[Idx] = Slider

        return Slider
    end

                    end
                end

                    Dropdown.Value = nil

                    Dropdown.Value = Val
                end

    function Funcs:AddDependencyBox()
        local Depbox = {Dependencies = {}}
        local Groupbox = self
        local Container = Groupbox.Container
        local Holder = Library.Create('Frame', {
            BackgroundTransparency = 1,
            Size = NewUDim2(1, 0, 0, 0),
            Visible = false,
            Parent = Container,
        })
        local Frame = Library.Create('Frame', {
            BackgroundTransparency = 1,
            Size = NewUDim2(1, 0, 1, 0),
            Visible = true,
            Parent = Holder,
        })
        local Layout = Library.Create('UIListLayout', {
            FillDirection = Enum.FillDirection.Vertical,
            SortOrder = Enum.SortOrder.LayoutOrder,
            Parent = Frame,
        })

        function Depbox:Resize()
            Holder.Size = NewUDim2(1, 0, 0, Layout.AbsoluteContentSize.Y)

            Groupbox:Resize()
        end

        Script.Function.Connection(Layout:GetPropertyChangedSignal('AbsoluteContentSize'), function()
            Depbox:Resize()
        end)
        Script.Function.Connection(Holder:GetPropertyChangedSignal('Visible'), function()
            Depbox:Resize()
        end)

        function Depbox:Update()
            for _, Dependency in next, Depbox.Dependencies do
                local Elem = Dependency[1]
                local Expected = Dependency[2]
                local CurrentValue = Elem.Value

                if CurrentValue == nil then
                    Holder.Visible = false

                    Depbox:Resize()

                    return
                end
                if CurrentValue ~= Expected then
                    Holder.Visible = false

                    Depbox:Resize()

                    return
                end
            end

            Holder.Visible = true

            Depbox:Resize()
        end
        function Depbox:SetupDependencies(Dependencies)
            for _, Dependency in next, Dependencies do
                assert(type(Dependency) == 'table', 'SetupDependencies: Dependency is not of type `table`.')
                assert(Dependency[1], 'SetupDependencies: Dependency is missing element argument.')
                assert(Dependency[2] ~= nil, 'SetupDependencies: Dependency is missing value argument.')
            end

            Depbox.Dependencies = Dependencies

            Depbox:Update()
        end

        Depbox.Container = Frame

        setmetatable(Depbox, BaseGroupbox)
        TableInsert(Library.DependencyBoxes, Depbox)

        return Depbox
    end

    BaseGroupbox.__index = Funcs
    BaseGroupbox.__namecall = function(Table, Key, ...)
        return Funcs[Key](...)
    end

        function Tab:HideTab()
            Blocker.BackgroundTransparency = 1
            TabButton.BackgroundColor3 = Library.BackgroundColor
            Library.RegistryMap[TabButton].Properties.BackgroundColor3 = 'BackgroundColor'
            TabFrame.Visible = false
        end
        function Tab:SetLayoutOrder(Position)
            TabButton.LayoutOrder = Position

            TabListLayout:ApplyLayout()
            UpdateTabSizes()
        end
        function Tab:AddGroupbox(Info)
            local Groupbox = {}
            local BoxOuter = Library.Create('Frame', {
                BackgroundColor3 = Library.BackgroundColor,
                BorderColor3 = Library.OutlineColor,
                BorderMode = Enum.BorderMode.Inset,
                Size = NewUDim2(1, 0, 0, 327 + 2),
                ZIndex = 2,
                Parent = Info.Side == 1 and LeftSide or RightSide,
            })

            Library.AddToRegistry(BoxOuter, {
                BackgroundColor3 = 'BackgroundColor',
                BorderColor3 = 'OutlineColor',
            })

            local BoxInner = Library.Create('Frame', {
                BackgroundColor3 = Library.BackgroundColor,
                BorderColor3 = C3_BLACK,
                Size = NewUDim2(1, -2, 1, -2),
                Position = NewUDim2(0, 1, 0, 1),
                ZIndex = 4,
                Parent = BoxOuter,
            })

            Library.AddToRegistry(BoxInner, {
                BackgroundColor3 = 'BackgroundColor',
            })

            local Highlight = Library.Create('Frame', {
                BackgroundColor3 = Library.AccentColor,
                BorderSizePixel = 0,
                Size = NewUDim2(1, 0, 0, 2),
                ZIndex = 5,
                Parent = BoxInner,
            })

            Library.AddToRegistry(Highlight, {
                BackgroundColor3 = 'AccentColor',
            })
            Library.CreateLabel({
                Size = NewUDim2(1, 0, 0, 18),
                Position = NewUDim2(0, 4, 0, 2),
                TextSize = 14,
                Text = Info.Name,
                TextXAlignment = Enum.TextXAlignment.Left,
                ZIndex = 5,
                Parent = BoxInner,
            })

            local Container = Library.Create('Frame', {
                BackgroundTransparency = 1,
                Position = NewUDim2(0, 4, 0, 20),
                Size = NewUDim2(1, -4, 1, -20),
                ZIndex = 1,
                Parent = BoxInner,
            })

            Library.Create('UIListLayout', {
                FillDirection = Enum.FillDirection.Vertical,
                SortOrder = Enum.SortOrder.LayoutOrder,
                Parent = Container,
            })

            function Groupbox:Resize()
                local Size = 0

                for _, Element in next, Groupbox.Container:GetChildren()do
                    if (not Element:IsA('UIListLayout')) and Element.Visible then
                        Size = Size + Element.Size.Y.Offset
                    end
                end

                BoxOuter.Size = NewUDim2(1, 0, 0, 20 + Size + 2 + 2)
            end

            Groupbox.Container = Container

            setmetatable(Groupbox, BaseGroupbox)
            Groupbox:AddBlank(3)
            Groupbox:Resize()

            Tab.Groupboxes[Info.Name] = Groupbox

            return Groupbox
        end
        function Tab:AddLeftGroupbox(Name)
            return Tab:AddGroupbox({
                Side = 1,
                Name = Name,
            })
        end
        function Tab:AddRightGroupbox(Name)
            return Tab:AddGroupbox({
                Side = 2,
                Name = Name,
            })
        end

        Script.Function.Connection(TabButton.InputBegan, function(Input)
            if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                Tab:ShowTab()
            end
        end)

        if #TabContainer:GetChildren() == 1 then
            Tab:ShowTab()
        end

        Window.Tabs[Name] = Tab

        task.defer(UpdateTabSizes)

        return Tab
    end

    local ModalElement = Library.Create('TextButton', {
        BackgroundTransparency = 1,
        Size = NewUDim2(0, 0, 0, 0),
        Visible = true,
        Text = '',
        Modal = false,
        Parent = ScreenGui,
    })
    local Toggled = false
    local Fading = false

    Library.Toggle = function()
        if Fading then
            return
        end

        local FadeTime = Config.MenuFadeTime

        Fading = true
        Toggled = not Toggled
        ModalElement.Modal = Toggled

        local blurGoal = Toggled and Library.BlurSize or 0
        local fovGoal = Toggled and (Library.FieldOfView - 12) or Library.FieldOfView

        TweenService:Create(Blur, TweenInfo.new(0.25, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {Size = blurGoal}):Play()
        TweenService:Create(CurrentCam, TweenInfo.new(0.25, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {FieldOfView = fovGoal}):Play()

        if Toggled then
            Outer.Visible = true
        end

        local fade = Outer:FindFirstChild('__fade')

        if not fade then
            fade = NewInstance('Frame')
            fade.Name = '__fade'
            fade.Size = UDim2Scale(1, 1)
            fade.BackgroundColor3 = C3_BLACK
            fade.BorderSizePixel = 0
            fade.ZIndex = 999
            fade.BackgroundTransparency = 1
            fade.Parent = Outer
        end

        TweenService:Create(fade, TweenInfo.new(FadeTime, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
            BackgroundTransparency = Toggled and 1 or 0.45,
        }):Play()

        for _, obj in pairs(Outer:GetDescendants())do
            if obj:IsA('TextLabel') or obj:IsA('TextBox') then
                obj.TextTransparency = Toggled and 0 or 1
            elseif obj:IsA('UIStroke') then
                obj.Transparency = Toggled and 0 or 1
            end
        end

        TaskDelay(FadeTime, function()
            Outer.Visible = Toggled
            Fading = false
        end)
    end

    Library.GiveSignal(Script.Function.Connection(InputService.InputBegan, function(Input, Processed)
        if type(Library.ToggleKeybind) == 'table' and Library.ToggleKeybind.Type == 'KeyPicker' then
            if Input.UserInputType == Enum.UserInputType.Keyboard and Input.KeyCode.Name == Library.ToggleKeybind.Value then
                TaskSpawn(Library.Toggle)
            end
        elseif Input.KeyCode == Enum.KeyCode.RightControl or (Input.KeyCode == Enum.KeyCode.RightShift and (not Processed)) then
            TaskSpawn(Library.Toggle)
        end
    end))

    if Config.AutoShow then
        TaskSpawn(Library.Toggle)
    end

    Window.Holder = Outer

    return Window
end)

OnPlayerChange = LPH_NO_VIRTUALIZE(function()
    local PlayerList = GetPlayersString()

    for _, Value in next, Options do
        if Value.Type == 'Dropdown' and Value.SpecialType == 'Player' then
            Value:SetValues(PlayerList)
        end
    end
end)

Script.Function.Connection(Players.PlayerAdded, OnPlayerChange)
Script.Function.Connection(Players.PlayerRemoving, OnPlayerChange)

Library.Toggles = Toggles
Library.Options = Options

getgenv().Library = Library
return Library
